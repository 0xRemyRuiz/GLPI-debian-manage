<VirtualHost *:80>
    ServerName _SERVER_NAME_
    ServerAlias _SERVER_LOCAL_IP_
    DocumentRoot /var/www/html/_GLPI_FOLDER_NAME_/public/

    Alias _URL_ /var/www/html/_GLPI_FOLDER_NAME_/public

    <Directory /var/www/html/_GLPI_FOLDER_NAME_/public>
        Options -Indexes +FollowSymLinks
        Require all granted
        RewriteEngine On

        RewriteRule ^ index.php [QSA,L]
    </Directory>

    <FilesMatch \.php$>

    </FilesMatch>

    CustomLog ${APACHE_LOG_DIR}/_GLPI_FOLDER_NAME__access.log combined
